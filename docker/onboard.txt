; New-client onboarding, run once at startup via --script-run (run-client.sh wires this in for a new
; game). It skips the character-creation intro, drops the player into Seyda Neen, and hands over a
; basic ready-to-play character so a freshly connected client is immediately in the shared world.
; Edit freely. (Ignored when you join with --save: the save already has a character and position.)

; Complete character generation. Without this the player is left in the half-created pre-chargen
; state (controls disabled, attributes/skills won't stick) even after leaving the intro cell.
; ChargenState -1 marks chargen done; then re-enable the controls the intro switched off.
set ChargenState to -1
EnablePlayerControls
EnablePlayerFighting
EnablePlayerJumping
EnablePlayerMagic
EnablePlayerViewSwitch

; New game disables all the GUI windows (disallowAll); chargen normally re-enables them step by step.
; Since we skip chargen, turn the playing UI back on ourselves (inventory, magic, map, stats, rest,
; level-up) — otherwise the player has no usable interface.
EnableInventoryMenu
EnableMagicMenu
EnableMapMenu
EnableStatsMenu
EnableRest
EnableLevelupMenu

; Leave the intro and arrive in Seyda Neen.
CenterOnCell "Seyda Neen"

; A modest starting character so the player isn't a blank slate.
player->SetStrength 40
player->SetIntelligence 40
player->SetWillpower 40
player->SetAgility 40
player->SetSpeed 40
player->SetEndurance 40
player->SetPersonality 40
player->SetLuck 40
player->SetHealth 100
player->SetMagicka 100
player->SetFatigue 200

; Starter kit: clothes, a blade, and a little gold.
player->AddItem "common_shirt_01" 1
player->AddItem "common_pants_01" 1
player->AddItem "common_shoes_01" 1
player->AddItem "iron_dagger" 1
player->AddItem "gold_001" 100
player->Equip "common_shirt_01"
player->Equip "common_pants_01"
player->Equip "common_shoes_01"
player->Equip "iron_dagger"
